TARGETS=(
  "$HOME/the_things_os/targets"
)

# 스냅샷 ID: 시간 + 랜덤
TS="$(date -u +%Y%m%dT%H%M%SZ)"
RID="$(cat /proc/sys/kernel/random/uuid)"
SID="${TS}__${RID}"

OUT="$SNAPDIR/$SID"
mkdir -p "$OUT"

# 1) 대상 복사 (no delete only add: 새 폴더에만 기록)
# rsync가 없으면 cp -a로 바꿔도 됨
for T in "${TARGETS[@]}"; do
  if [ -e "$T" ]; then
    rsync -a --numeric-ids --delete-excluded --exclude ".git/" "$T" "$OUT/" 2>/dev/null || cp -a "$T" "$OUT/"
  fi
done

# 2) 메타데이터 기록
META="$OUT/metadata.json"
cat > "$META" <<EOF
{
  "snapshot_id": "$SID",
  "created_at_utc": "$TS",
  "principle": "no_delete_only_add",
  "type": "file_snapshot",
  "targets": $(printf '%s\n' "${TARGETS[@]}" | python3 - <<'PY'
import json,sys
print(json.dumps([l.strip() for l in sys.stdin if l.strip()], ensure_ascii=False, indent=2))
PY
)
}
EOF

# 3) 무결성(선택) 해시 목록 생성: 나중에 깨졌는지 확인 가능
# 파일이 많으면 오래 걸릴 수 있음. 일단 켜두고, 느리면 주석 처리하면 됨.
(
  cd "$OUT"
  find . -type f -name "*.json" -o -type f -maxdepth 4 | sort | xargs -r sha256sum > manifest.sha256 2>/dev/null || true
)

ln -sfn "$OUT" "$BASE/run/current_stable"

# 5) 로그
echo "[$(date -u +%Y-%m-%dT%H:%M:%SZ)] SNAPSHOT $SID -> $OUT" | tee -a "$LOGDIR/snapshot.log"

id

관리자는 꼭 root일 필요는 없다.
sudo 그룹에 속해 있으면 관리자 기능 가능.

확인:

groups
또는

sudo -l
3️⃣ 실제 시스템은 이렇게 판단한다
구분	판단 기준
절대 관리자	UID == 0
위임 관리자	sudo 권한 있음
일반 사용자	sudo 없음

실제 SSH/관리 포털: 토큰 검증

토큰은 예를 들면:

TOTP(시간 기반)

또는 서명된 짧은 수명 토큰(JWT 같은 것)

여기서 핵심은 짧은 TTL과 좁은 권한(scope)

B. 절대관리자도 예외 없음 (Zero Trust)
대표/보안책임자도 같은 절차

대신 “긴급 모드”는 별도 경로로, 감사 로그 100% 남김

C. “날짜 변화 + uuid OTP”는 TOTP로 정식화
너가 말한 “날짜 변화에 따른 OTP”는 사실상:

time_step (예: 30초, 60초)

secret_key

HMAC

으로 정리되는 TOTP가 정석이야.

너의 “uuid type”은

seed(시크릿)로 쓰거나

사용자별 키 식별자(kid)로 쓰는 건 OK

하지만 “uuid 자체를 OTP처럼” 쓰면 위험해질 수 있음(예측성/관리 문제)

4) 단독 서버 패턴변환기를 더 안전하게 만드는 규칙
✅ 1) 변환 규칙은 “공개”되어도 안전해야 함
보안은 규칙이 아니라 키에서 나온다 (Kerckhoffs 원칙).

✅ 2) 키는 파일이 아니라 가능하면 HSM/TPM/보안영역
최소한:

root만 읽는 파일(600) + 별도 디스크 암호화

키 로테이션

✅ 3) 토큰은 짧게, 권한은 좁게
TTL 30~120초

scope: “ssh to bastion only”, “read-only only” 같은 제한

✅ 4) 감사 로그는 중앙에 append-only
발급 로그

실패 로그

재시도 패턴

IP/ASN/지리(가능하면)

5) “관리자가 규정 어기고 접속”을 어떻게 잡나?
여기서 패턴검사는 암호가 아니라 정책/행동 분석이야.

잡는 포인트:

접속 시간대(근무시간/비상시간)

접속 경로(허용 IP/디바이스)

실패 횟수/속도

발급 토큰 요청 빈도

평소와 다른 지역/ASN

“토큰 발급 요청”은 있었는데 “실제 접속”이 없는 이상 패턴

이건 보안팀이 하는 UEBA(행동 분석)랑 결이 같다.

1️⃣ 대표 ≠ 보안책임자
기업 구조에서 역할은 이렇게 나뉜다:

대표(CEO) → 방향·의사결정

보안책임자(CISO) → 위험 관리·통제 설계

시스템 관리자 → 기술 운영

감사/내부통제 → 검증·감시

이걸 섞는 순간 보안은 무너진다.

왜냐하면:

권한 집중 = 통제 붕괴

2️⃣ 절대권한은 설계 실패의 신호
보안에서 가장 위험한 구조는:

“절대관리자 1인”

이 구조는:

내부 위협에 취약

실수에 취약

감정에 취약

감사 불가능

그래서 현대 보안은 이렇게 설계한다:

2인 승인

역할 분리

최소 권한

시간 제한 권한

모든 행위 로그

3️⃣ Cloud IDC 목표라면 더 엄격해야 한다
Cloud 레벨에서는:

대표도 root 접근 불가

CISO도 직접 서버 접속 불가

운영자는 역할 기반 접근

긴급 접근은 승인 + 로그 100%

이게 정상이다.

4️⃣ 네가 말한 “패턴 변환기”와 연결해보면
패턴 변환기가 진짜 역할을 하려면:

대표 코드

보안책임자 코드

시스템 관리자 코드

이걸 같은 계층에 두면 안 된다.

대표는 정책 승인 권한
보안책임자는 통제 승인 권한
운영자는 실행 권한

각각 분리해야 한다.

5️⃣ 진짜 중요한 원칙

신뢰 기반이 아니라
통제 기반이다.

대표가 아무리 선해도
구조는 그걸 가정하면 안 된다.
